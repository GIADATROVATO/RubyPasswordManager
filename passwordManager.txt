
FILE_SALVATAGGIO = “credenziali.dat"
# ————————————
#	MASTER PASSWORD
# ————————————
def inserisci_chiave
	print ”Inserisci password manager”
	STDIN.noecho(&:gets).chomp.tap {puts} 
end
# ——————————————————
#	INSERIMENTO CREDENZIALE 
# ——————————————————
Def inserisici_credenziale 
	print “Inserisci il sito: “ 
	sito = gets.chomp 
	print “Inserisci il nome utente: “
	nome_utente= gets.chomp 
	print “Inserisci password: ”
	password= STDIN.noecho(&:gets).chomp 
	puts 
	{
	sito: sito,
	nome_utente: nome_utente, 
	password: password 
	}
end	
# ————————————
#	VISUALIZZAZIONE
# ————————————
def visualizza_credenziale(credenziale) 
	puts "\nLe tue credenziali:"
	puts " — Sito:  #{credenziale[:sito]}"
	puts " — Nome: #{credenziali[:nome_utente]}”
	puts " — Password: #{credenziali[:password]}” 
end 
# ————————————
#	MENU
# ————————————
def mostra_menu 
	puts "\nPASSWORD MANAGER” 
	puts "1. Aggiungi credenziale"
	puts "2. Mostra credenziali"
	puts "3. Esci"
	print "Scelta: "
end

# ————————————
#	CIFRATURA
# ————————————
def crea_cipher(mode, chiave)
	cipher= OpenSSL::Cipher.new("AES-256-CBC”)
	mode == :encrypt ? cipher.encrypt : cipher.decrypt 
	cipher.key=Digest::SHA256.digest(chiave) 
	cipher 							
end

# ————————————
#	SALVATAGGIO FILE
# ————————————
			
def salva_credenziali( credenziali, chiave ) 
	dati = JSON.generate(credenziali)
/*
*	trasforma questo [{sito: "gmail”, nome_utente: “Mario”,password: "1234"}]
*					in 
*	"[{\"sito\":\"gmail\",\"nome_utente\":\"mario\",\"password\":\"1234\"}]"
*
*	perche AES cifra stringhe di byte e non oggetti ruby 
*/
	cipher=  crea_cipher(:encrypt, chiave) 
	iv= cipher.random_iv 
/*
*	IV= initializer vector , è un numero casuale di 16 byte 
*			(per AES, che ha blocchi da 128 bit=16 bytes)
*/
	encrypted= cipher.update(dati) + cipher.final 
	File.binWrite(FILE_SALVATAGGIO , iv + encrypted)
/*
*	salvo nel file [ IV ] [ DATI CIFRATI ] perché quando leggo il file dovrò
*	1. prendere i primi 16byte (IV) 
*	2. Usare il resto per decifrare 
/*
end
# ————————————
#	CARICAMENTO FILE
# ————————————
Def carica_credenziali(chiave) 
	return [ ] unless File.exists?(FILE_SALVATAGGIO) 
	contenuto = File.binread(FILE_SALVATAGGIO) 

	iv= contenuto[0,16]
  dati_criptati= contenuto[16..]

	cipher=  crea_cipher(:decrypt, chiave) 
	cipher.iv=iv	

	decrypted= cipher.update( dati_criptati) + cipher.final 	
	JSON.parse( decrypted, symbolize_names: true)

rescue 
	puts “Password errata o file corrotto” 
	[ ]	
end
# ————————————————
#	PROGRAMMA PRINCIPALE
# ————————————————
chiave= inserisci_chiave
credenziali= carica_credenziali(chiave)
loop do 
	mostra_menu
	scelta= gets.chomp
	case scelta 
	when “1” 
		credenziale= inserisici_credenziale
		credenziali << credenziale 
		salva_credenziali( credenziali, chiave )
	when “2”
		credenziali.each { |c| visualizza_credenziale(c) }

	when “3” 
		salva_credenziali( credenziali, chiave )
		puts “Arrivederci!”
		break 
	else 
		puts “Scelta non valida”
	end
end
